###### Problem p1
p1c1 = RandInt(-10,10)
p1c2 = RandInt(-10,10)
p1c3 = NonZeroInt(-5,5)
p1c4 = RandInt(1,10)
p1f1 = p1c3*x + p1c1
p1f2 = p1c3*x - p1c1
p1f3 = p1c4*x + p1c2

p1f4 = expand(p1f1*p1f2*p1f3)
