##### Problem p1
p1b1 = RandInt(2,7)
p1arg1 = 9999
while p1arg1 > 500:
    p1pwr1 = RandInt(0,5)
    p1arg1 = p1b1^p1pwr1
    p1arg2 = p1b1^(-p1pwr1)
p1ans1 = p1pwr1
p1ans2 = -p1pwr1
