###### Problem p1
p1b1 = RandInt(2,7)

p1c1 = NonZeroInt(-5,5)
p1c2 = RandInt(-5,5)
p1c3 = RandInt(-5,5)

p1f1 = p1c1*x + p1c2

p1ans1 = (p1b1^p1c3-p1c2)/p1c1

\begin{problem}
    Solve for $x$ in the following logarithmic equation:
    \[
        \log_{\sage{p1b1}}\left(\sage{p1f1}\right) = \sage{p1c3}
    \]

    $x = \answer{\sage{p1ans1}}$.
\end{problem}
