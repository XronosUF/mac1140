\begin{problem}
    Consider the following graph:
    \begin{center}
        \begin{tikzpicture}
            \begin{axis}[
                axis x line=middle, 
                axis y line=middle, 
                xlabel={$x$}, 
                ylabel={$y$}, 
                xtick={-4,-2,...,10}, 
                ytick={-6,-4,...,10},
                xmin=-3, 
                xmax=10, 
                ymin=-5, 
                ymax=10
                ]
                \addplot[<->,domain=-2.9:9, samples=300]{-2*ln(x+3) + 4};
            \end{axis}
        \end{tikzpicture}
    \end{center}
    
    What is the best parent function for the graph?
    
    \begin{multipleChoice}
        \choice{$f(x)=x$}
        \choice{$f(x)=x^2$}
        \choice{$f(x)=x^3$}
        \choice{$f(x)=\sqrt{x}$}
        \choice{$f(x)=e^x$}
        \choice[correct]{$f(x)=\ln(x)$}
        \choice{$f(x)=|x|$}
    \end{multipleChoice}
    
    
\end{problem}
